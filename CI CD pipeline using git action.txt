name: CI/CD Pipeline

on:
  push:
    branches:
      - main

jobs:
  build-deploy:
    runs-on: ubuntu-latest

    steps:
    - name: Checkout Code
      uses: actions/checkout@v3

    - name: Build Step
      run: echo "Building project..."

    - name: Test Step
      run: echo "Testing project..."

    - name: Deploy Step
      run: echo "Deploying project..."
